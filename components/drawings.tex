% Usage:
%   \drawing[height=2cm]{a01-alice}          % any file from simon-drawings/, no extension
%   \drawingnode{(x,y)}{label}{a01-alice}    % drop a drawing at a TikZ coordinate

\newcommand{\drawing}[2][height=1.2cm]{%
  \includegraphics[#1]{simon-drawings/#2}%
}

% Place a drawing as a TikZ node.
% Args: [img options] {node name} {coordinate} {file stem}
% Example: \drawingnode[height=1.2cm]{alice}{(0,0)}{a01-alice}
\newcommand{\drawingnode}[4][height=1.2cm]{%
  \node (#2) at #3 {\includegraphics[#1]{simon-drawings/#4}};%
}
